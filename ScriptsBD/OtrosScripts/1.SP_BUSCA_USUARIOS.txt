CREATE PROCEDURE SP_OBTIENE_USUARIO
(
@NO_USUA VARCHAR(6)
)
AS
BEGIN
	SELECT U.CO_USUA,
			U.NO_USUA,
			U.US_CREA,
			U.US_MODI,
			U.FE_CREA,
			U.FE_MODI,
			P.CO_PERS,
			P.NU_DOCU,
			P.NO_PERS,
			P.AP_PATE,
			P.AP_MATE,
			P.SEXO,
			P.FE_NACI,
			P.US_CREA,
			P.US_MODI,
			P.FE_CREA,
			P.FE_MODI
		FROM USUA U
			INNER JOIN PERS P
				ON U.CO_PERS = P.CO_PERS
		WHERE U.NO_USUA LIKE @NO_USUA + '%' 
END
